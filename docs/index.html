from specops import SpecOps

# Initialize SpecOps with your preferred AI provider
specops = SpecOps(
    provider="openai",
    api_key="your-api-key"
)

# Analyze any GitHub repository
analysis = specops.analyze_repository(
    url="https://github.com/facebook/react"
)

# Generate comprehensive onboarding materials
docs = specops.generate_all_documents(analysis)

print(f"Generated {len(docs)} onboarding documents")
# Output: Generated 3 onboarding documents
